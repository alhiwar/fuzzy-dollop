قصيدة 
" يا سيدي الحسين "
  عُـصِـمْـتَ فـلـيـس تـقـربُـكَ الـظـنـون ُ
كـأنــكَ لــلــهُـــدى والــدِّيــنِ دِيــن ُ
//
فــيــا مَــنْ " حـاؤهُ " حَـقٌّ وحِــلــم ٌ
ويـا مَـنْ " سِـيـنُـهُ " مُـنْـج ٍ سَــفـيـن ُ
//
ويـا مَــنْ " يـاؤهُ " يُــسْـــر ٌ ويَــمٌّ
ويـا مَـنْ " نـونُـه ُ " نـورٌ مُــبـــيــن ُ
//
ويا ابْـنَ (الـدُّرِّ والـذّهَـب ِ الـمُـصـفّى)
ومَــنْ أوصـى بـبـيـعـتِـهِ " الأمــيــنُ " (1)
//
ويــا مَــنْ حُــبُّــهُ تــقــوىً ورِبْــحٌ
ويـا مَـنْ بُـغـضُـهُ خُــسـرٌ مُـشِــيـنُ
//
ويــا مَـنْ أعْـجَــزَ الــدُّنــيــا بــبَــذل ٍ
غـداةَ اسْـتُـهْـدِفَ " الـحَـبْـلُ الـمـتـيـنُ " (2)
//
ويـا ابْـنَ الأطـهَــرَيـن ِ : أبــا ً وأمّــا ً
وتـخــشـى نـورَ غُـرَّتِـه ِ الــدُّجـونُ (3)
//
ويـا مَـنْ لــن يـكـون لــهُ شـــبــيـهٌ
عـلـى الأرضـيـن مـامـرَّتْ ســنـونُ
//
حَــبـاهُ الــلــهُ دون الـخــلــق ِ أمــرا ً
يــدومُ الــدّهـــرَ مـادامَ الــيــقـــيــنُ
//
هـو الــشْــمـسـان ِ لـكـنْ لا غــروب ٌ
لــنــورِهِــمـا ومـا لـهــمـا قــريـنُ
//
ويــا مَــنْ لا يُـــرَدُّ لــهُ دعـــاءٌ
ويـسْــتـرجـي شــفـاعــتَـهُ الـمـكـيـنُ
//
إذا ذُكِــرَ الـحـســـيـنُ أصـاخَ كــون ٌ
وجــلـجـل فـي تـهــجّــدِهِ الـسـكـونُ
//
فـدىً لــتـرابِ نـعـلِــكَ يـا إمـامـي :
أنـا .. وأبـي .. وأمـي .. والــبـنـونُ
//
أخـفـتَ الـمـوتَ حـتـى خـرَّ ذُعــرا ً
فـأنـتَ الـحيُّ والــمــوتُ الـدَّفــيــنُ
//
وأنـتَ لــكـلِّ ذي عــزم ٍ حــســـامٌ
وأنـت لـكـلِّ مــذعــور ٍ حُـصــونُ
//
أبـا الأحـرار هـلاّ قــمْـتَ فــيـنـا ؟
فــقــد عـمَّ الــبـلاءُ .. ولا مُـعــيـنُ
//
ولا " حُــرٌّ ريــاحـيٌّ " فـيُـفـدي ..
ولا الـعـبّـاسُ والـقـدّيــسُ " جُـونُ " (4)
//
تـأسّــدَتِ الأرانــبُ .. فـالــرّزايـا
حُــبـالـى والــنـوائــبُ والــمـنـونُ
//
تـقـنّـعـتِ الـوجـوهُ فـلـيـس نـدري
أشــوكٌ حـولــنــا ؟ أمْ يـاســمـيـنُ ؟
//
" يـزيـدٌ " بات هـذا الـعـصـرَ جـمعا ً
يُــنـاصِـرُهُ الـمُـهَــتَّــكُ والـلـعــيـنُ
//
وفـيـنـا ألفُ " حرمـلة ٍ" و" شـمــرٍ "
تـفـخَّـخَ فــيـهــمُ الـحـقــدُ الـدَّفــيـنُ
//
وفـيـنـا مـن " أبي جهـل ٍ" عـشـيـرٌ
ولـ "ابْن زيادَ " من جـذرٍ غـصـونُ
//
وفــيــنـا آكـلـو الأكـبـادِ غـصّـتْ
بـهـم دارُ الـخـلافـةِ لا الــسـجـونُ (5)
//
حُـسـيـنـيـون إنْ نـطـقـوا .. وأمّـا
فــعـالــهُــمُ فـأبــعــدُ مـا تــكــونُ
**
(1) المقبوس من البيت الشهير المنسوب إلى علي بن ابراهيم الكفعمي :
علي الدرُّ والذهب المصفى
وباقي الناس كلهم ترابُ
(2) الحبل المتين : الدين الإسلامي كما في قول الله تعالى : " واعتصموا بحبل الله " ..
(3) الدجون : ظلام الغيم يسود الأرض .. ومن معانيه العتمة الشديدة .
(4) جون : هو جون بن حوي ـ مولى الصحابي الجليل أبي ذر الغفاري ـ وقد استشهد مع الحسين عليه السلام.
(5) إشارة إلى بعض القتلة الذين سبق وتصدروا واجهة المشهد السياسي العراقي ضمن نظام المحاصصة المقيت حيث أثبتت التحقيقات واعترافات المجرمين ضلوعهم في الكثير من الأعمال الإرهابية والتفجيرات والخطف والقتل والإغتصاب
